        // --- 1. HERO SCROLL EFFECT ---
        const sideTexts = document.querySelectorAll('.side-text');
        window.addEventListener('scroll', () => {
            const scrollY = window.scrollY;
            sideTexts.forEach(text => {
                if(scrollY < window.innerHeight) { text.style.transform = `translateY(-${scrollY * 0.5}px)`; }
            });
        });

        // --- 2. TYPEWRITER EFFECT ---
        const typingElement = document.querySelector('.typing-text');
        const roles = ["Electronics Engineer", "AI/ML Engineer", "Graphic Designer"];
        let roleIndex = 0, charIndex = 0, isDeleting = false;
        
        function typeEffect() {
            const currentRole = roles[roleIndex];
            if (isDeleting) { typingElement.textContent = currentRole.substring(0, charIndex - 1); charIndex--; }
            else { typingElement.textContent = currentRole.substring(0, charIndex + 1); charIndex++; }
            let nextSpeed = isDeleting ? 50 : 100;
            if (!isDeleting && charIndex === currentRole.length) { isDeleting = true; nextSpeed = 2000; }
            else if (isDeleting && charIndex === 0) { isDeleting = false; roleIndex = (roleIndex + 1) % roles.length; nextSpeed = 500; }
            setTimeout(typeEffect, nextSpeed);
        }
        document.addEventListener('DOMContentLoaded', typeEffect);

        // --- 3. PROJECT SLIDER LOGIC WITH CENTER DETECTION & BLINK ---
        const slider = document.getElementById('projectsSlider');
        const numbersContainer = document.getElementById('sliderNumbers');
        const cards = document.querySelectorAll('.project-card');

        // Generate Numbers
        cards.forEach((card, index) => {
            const number = document.createElement('div');
            number.classList.add('slider-number');
            number.textContent = index + 1; 
            
            // CLICK EVENT: Center the card & blink BOTH
            number.addEventListener('click', () => {
                const scrollPos = card.offsetLeft - (slider.offsetWidth / 2) + (card.offsetWidth / 2);
                
                slider.scrollTo({
                    left: scrollPos,
                    behavior: 'smooth'
                });

                // Reset Animations
                number.classList.remove('blink-active');
                card.classList.remove('flash-active');
                
                // Force Reflow
                void number.offsetWidth;
                void card.offsetWidth; 
                
                // Add Active Classes
                number.classList.add('blink-active');
                card.classList.add('flash-active');
                
                // Remove after animation ends
                setTimeout(() => {
                    number.classList.remove('blink-active');
                    card.classList.remove('flash-active');
                }, 800);
            });
            
            numbersContainer.appendChild(number);
        });

        const numberEls = document.querySelectorAll('.slider-number');

        // SCROLL LISTENER: Active State based on Center Proximity
        slider.addEventListener('scroll', () => {
            const sliderCenter = slider.scrollLeft + (slider.offsetWidth / 2);
            
            let closestCardIndex = 0;
            let minDistance = Infinity;

            cards.forEach((card, index) => {
                const cardCenter = card.offsetLeft + (card.offsetWidth / 2);
                const distance = Math.abs(sliderCenter - cardCenter);
                
                if (distance < minDistance) {
                    minDistance = distance;
                    closestCardIndex = index;
                }
            });

            numberEls.forEach((num, index) => {
                if (index === closestCardIndex) {
                    num.classList.add('active');
                } else {
                    num.classList.remove('active');
                }
            });
        });
        
        // Trigger scroll event once on load to highlight #1
        slider.dispatchEvent(new Event('scroll'));

    </script>
